//1 åˆ° 100 000
#include <iostream>
#include <math.h>

using namespace std;
#define N 100000
typedef long long LL;

LL a;
LL b;
LL vec[N];
int cnt;
LL  MAX_MUL;

void Find(const LL* vec)
{
    int sum=0;
    LL  mul=1;
    LL  Now=1;
    for(int i=0;i<cnt;i++)
    {
        sum+=vec[i];
        while(mul%vec[i]!=0)
            mul*=(++Now);
        mul/=vec[i];
    }
    while(Now<100000)
         mul*=(++Now);
    LL diff=((1+N)*N)/2-sum;
    cout<<diff<<" "<<mul<<endl;
    LL a=(diff+sqrt(diff*diff-4*mul))/2;
    cout<<a<<" "<<diff-a<<endl;
}

int main()
{
    srand(time(NULL));
    a=(rand()%100000)+1;
    b=(rand()%100000)+1;
    cnt=0;
    for(int i=1;i<=N;i++)
    {
        if(i!=a&&i!=b)
            vec[cnt++]=i;
    }
    cout<<a<<" "<<b<<" "<<endl;
    cout<<a+b<<" "<<a*b<<endl;
    Find(vec);
}
